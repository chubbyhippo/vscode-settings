:raw_base_settings_url: https://raw.githubusercontent.com/chubbyhippo/vscode-settings/refs/heads/main

= Install settings & keymappings

== curl

[source,shell,subs="attributes"]
----
curl {raw_base_settings_url}/install.sh | /usr/bin/env sh
----

== wget

[source,shell,subs="attributes"]
----
wget -qO - {raw_base_settings_url}/install.sh | /usr/bin/env sh
----

== powershell

[source,powershell,subs="attributes"]
----
irm {raw_base_settings_url}/install.ps1 | iex
----

== Install default extensions

[source,shell,subs="attributes"]
----
curl {raw_base_settings_url}/install-default-extensions.sh | /usr/bin/env sh
----

[source,powershell,subs="attributes"]
----
Invoke-RestMethod {raw_base_settings_url}/install-extensions.sh | Invoke-Expression
----

== Uninstall extensions

[source,shell]
----
code --list-extensions | xargs -L 1 code --uninstall-extension;code --list-extensions | xargs -L 1 code --uninstall-extension
----
